# PM2 Management
manage_pm2() {
    print_header
    print_msg "$CYAN" "📦 PM2 Process Manager"
    echo
    
    # Check if PM2 is installed
    if ! command_exists pm2; then
        print_msg "$RED" "❌ PM2 is not installed!"
        echo
        print_msg "$YELLOW" "Would you like to install PM2? (yes/no)"
        read -p "> " install_pm2
        
        if [ "$install_pm2" = "yes" ]; then
            print_msg "$BLUE" "📦 Installing PM2..."
            npm install -g pm2
            print_msg "$GREEN" "✅ PM2 installed successfully!"
        fi
        echo
        read -p "Press Enter to continue..."
        return
    fi
    
    echo -e "${YELLOW}1)${NC} List all PM2 processes"
    echo -e "${YELLOW}2)${NC} Restart all processes"
    echo -e "${YELLOW}3)${NC} Restart specific app"
    echo -e "${YELLOW}4)${NC} Stop all processes"
    echo -e "${YELLOW}5)${NC} Start all processes"
    echo -e "${YELLOW}6)${NC} View logs (real-time)"
    echo -e "${YELLOW}7)${NC} View app logs"
    echo -e "${YELLOW}8)${NC} Monitor processes"
    echo -e "${YELLOW}9)${NC} Save PM2 state"
    echo -e "${YELLOW}10)${NC} Delete process"
    echo -e "${YELLOW}11)${NC} Restart PM2 daemon"
    echo -e "${YELLOW}0)${NC} Back"
    echo
    read -p "Select option: " pm2_choice
    
    case $pm2_choice in
        1)
            pm2 list
            ;;
        2)
            print_msg "$BLUE" "🔄 Restarting all PM2 processes..."
            pm2 restart all
            pm2 save
            print_msg "$GREEN" "✅ All processes restarted!"
            log_action "PM2: Restarted all processes"
            ;;
        3)
            pm2 list
            echo
            read -p "Enter app name or ID: " app_name
            if [ -n "$app_name" ]; then
                pm2 restart "$app_name"
                print_msg "$GREEN" "✅ Process restarted!"
                log_action "PM2: Restarted $app_name"
            fi
            ;;
        4)
            pm2 stop all
            print_msg "$YELLOW" "⚠️  All processes stopped!"
            log_action "PM2: Stopped all processes"
            ;;
        5)
            pm2 start all
            print_msg "$GREEN" "✅ All processes started!"
            log_action "PM2: Started all processes"
            ;;
        6)
            print_msg "$CYAN" "📋 Real-time logs (Press Ctrl+C to exit)..."
            sleep 1
            pm2 logs
            ;;
        7)
            pm2 list
            echo
            read -p "Enter app name: " app_name
            if [ -n "$app_name" ]; then
                pm2 logs "$app_name" --lines 100
            fi
            ;;
        8)
            pm2 monit
            ;;
        9)
            pm2 save
            print_msg "$GREEN" "✅ PM2 state saved!"
            log_action "PM2: State saved"
            ;;
        10)
            pm2 list
            echo
            read -p "Enter app name or ID to delete: " app_name
            if [ -n "$app_name" ]; then
                pm2 delete "$app_name"
                pm2 save
                print_msg "$GREEN" "✅ Process deleted!"
                log_action "PM2: Deleted $app_name"
            fi
            ;;
        11)
            pm2 kill
            sleep 2
            pm2 resurrect
            print_msg "$GREEN" "✅ PM2 daemon restarted!"
            log_action "PM2: Daemon restarted"
            ;;
        0)
            return
            ;;
    esac
    
    echo
    read -p "Press Enter to continue..."
}

        try_files \$uri \$uri/ /index.html;
    }
    
    # Block access to sensitive files
    location ~ /\. {
        deny all;
    }

    # Block access to sensitive files

        deny all;
    }

            fi

        fi

# List all domains
list_domains() {
    print_header
    print_msg "$CYAN" "📋 Configured Domains"
    echo -e "${BLUE}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
    
    if [ ! -d "$NGINX_SITES_ENABLED" ]; then
        print_msg "$RED" "❌ Nginx sites-enabled directory not found"
        echo
        read -p "Press Enter to continue..."
        return
    fi
    
    local count=1
    local found=0
    
    for site in "$NGINX_SITES_ENABLED"/*; do
        if [ -f "$site" ] && [ ! -L "$site" -o -e "$site" ]; then
            local domain=$(basename "$site")
            local ssl_status="${RED}No SSL${NC}"
            
            if [ -d "$SSL_CERT_PATH/$domain" ]; then
                ssl_status="${GREEN}SSL Enabled${NC}"
            fi
            
            echo -e "${YELLOW}$count)${NC} $domain - $ssl_status"
            ((count++))
            found=1
        fi
    done
    
    if [ $found -eq 0 ]; then
        print_msg "$YELLOW" "⚠️  No configured domains found"
    fi
    
    echo -e "${BLUE}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
    echo
    read -p "Press Enter to continue..."
}

# SSL Management
manage_ssl() {
    print_header
    print_msg "$CYAN" "🔒 SSL Certificate Management"
    echo
    
    if ! command_exists certbot; then
        print_msg "$RED" "❌ Certbot is not installed!"
        echo
        print_msg "$YELLOW" "Install certbot with: apt install certbot python3-certbot-nginx"
        echo
        read -p "Press Enter to continue..."
        return
    fi
    
    echo -e "${YELLOW}1)${NC} Check certificate status"
    echo -e "${YELLOW}2)${NC} Renew all certificates"
    echo -e "${YELLOW}3)${NC} View certificate details"
    echo -e "${YELLOW}4)${NC} Test auto-renewal"
    echo -e "${YELLOW}5)${NC} Add SSL to domain"
    echo -e "${YELLOW}0)${NC} Back"
    echo
    read -p "Select option: " ssl_choice
    
    case $ssl_choice in
        1)
            print_msg "$BLUE" "🔍 Checking certificate status..."
            if [ ! -d "$SSL_CERT_PATH" ]; then
                print_msg "$RED" "❌ No certificates found"
                echo
                read -p "Press Enter to continue..."
                return
            fi
            
            local found=0
            for cert_dir in "$SSL_CERT_PATH"/*; do
                if [ -d "$cert_dir" ]; then
                    local domain=$(basename "$cert_dir")
                    if [ -f "$cert_dir/fullchain.pem" ]; then
                        local expiry=$(openssl x509 -enddate -noout -in "$cert_dir/fullchain.pem" 2>/dev/null | cut -d= -f2)
                        if [ -n "$expiry" ]; then
                            local days_left=$(( ($(date -d "$expiry" +%s 2>/dev/null || echo 0) - $(date +%s)) / 86400 ))
                            
                            if [ $days_left -gt 30 ]; then
                                echo -e "${GREEN}✅ $domain${NC}: $days_left days remaining"
                            elif [ $days_left -gt 0 ]; then
                                echo -e "${YELLOW}⚠️  $domain${NC}: $days_left days remaining"
                            else
                                echo -e "${RED}❌ $domain${NC}: EXPIRED!"
                            fi
                            found=1
                        fi
                    fi
                fi
            done
            
            if [ $found -eq 0 ]; then
                print_msg "$YELLOW" "⚠️  No valid certificates found"
            fi
            ;;
        2)
            print_msg "$BLUE" "🔄 Renewing certificates..."
            certbot renew && print_msg "$GREEN" "✅ Renewal successful!" || print_msg "$RED" "❌ Renewal failed!"
            systemctl reload nginx 2>/dev/null
            log_action "SSL: Certificates renewed"
            ;;
        3)
            read -p "Enter domain name: " domain
            if [ -n "$domain" ] && [ -f "$SSL_CERT_PATH/$domain/fullchain.pem" ]; then
                openssl x509 -in "$SSL_CERT_PATH/$domain/fullchain.pem" -text -noout 2>/dev/null | head -n 30
            else
                print_msg "$RED" "❌ Certificate not found for $domain"
            fi
            ;;
        4)
            print_msg "$BLUE" "🧪 Testing auto-renewal..."
            certbot renew --dry-run
            ;;
        5)
            read -p "Enter domain name: " domain
            if [ -n "$domain" ]; then
                certbot --nginx -d "$domain" -d "www.$domain"
                log_action "SSL: Added certificate for $domain"
            fi
            ;;
        0)
            return
            ;;
    esac
    
    echo
    read -p "Press Enter to continue..."
}

# Backup system
create_backup() {
    print_header
    print_msg "$CYAN" "💾 Create Backup"
    echo
    
    print_msg "$YELLOW" "Select backup type:"
    echo -e "${YELLOW}1)${NC} Full backup (Website + Nginx + SSL + PM2)"
    echo -e "${YELLOW}2)${NC} Website only"
    echo -e "${YELLOW}3)${NC} Configuration only"
    echo -e "${YELLOW}0)${NC} Cancel"
    echo
    read -p "Select option: " backup_choice
    
    if [ "$backup_choice" = "0" ]; then
        return
    fi
    
    local timestamp=$(date +%Y%m%d_%H%M%S)
    local backup_name="backup_${timestamp}"
    local backup_path="$BACKUP_DIR/$backup_name"
    
    mkdir -p "$backup_path"
    
    case $backup_choice in
        1)
            print_msg "$BLUE" "📦 Creating full backup..."
            
            print_msg "$YELLOW" "📁 Backing up website files..."
            if [ -d /var/www ]; then
                tar -czf "$backup_path/website.tar.gz" -C /var/www . 2>/dev/null && print_msg "$GREEN" "✅ Website backed up" || print_msg "$RED" "❌ Failed"            fi
            
            print_msg "$YELLOW" "⚙️  Backing up Nginx config..."
            if [ -d /etc/nginx ]; then
                tar -czf "$backup_path/nginx.tar.gz" -C /etc/nginx . 2>/dev/null && print_msg "$GREEN" "✅ Nginx backed up" || print_msg "$RED" "❌ Failed"
            fi
            
            print_msg "$YELLOW" "🔒 Backing up SSL certificates..."
            if [ -d /etc/letsencrypt ]; then
                tar -czf "$backup_path/ssl.tar.gz" -C /etc/letsencrypt . 2>/dev/null && print_msg "$GREEN" "✅ SSL backed up" || print_msg "$RED" "❌ Failed"            fi
            
            print_msg "$YELLOW" "📦 Backing up PM2 config..."
            if command_exists pm2; then
                pm2 save 2>/dev/null
                if [ -d ~/.pm2 ]; then
                    cp -r ~/.pm2 "$backup_path/pm2_config" 2>/dev/null && print_msg "$GREEN" "✅ PM2 backed up" || print_msg "$RED" "❌ Failed"
                fi
            fi
            
            print_msg "$YELLOW" "🔧 Backing up SSLH config..."
            if [ -f /etc/default/sslh ]; then
                cp /etc/default/sslh "$backup_path/" 2>/dev/null && print_msg "$GREEN" "✅ SSLH backed up" || print_msg "$RED" "❌ Failed"
            fi
            ;;
        2)
            print_msg "$BLUE" "📁 Backing up website only..."
            if [ -d /var/www ]; then
                tar -czf "$backup_path/website.tar.gz" -C /var/www . 2>/dev/null && print_msg "$GREEN" "✅ Success" || print_msg "$RED" "❌ Failed"
            else
                print_msg "$RED" "❌ Website directory not found"
            fi
            ;;
        3)
            print_msg "$BLUE" "⚙️  Backing up configurations only..."
            if [ -d /etc/nginx ]; then
                tar -czf "$backup_path/nginx.tar.gz" -C /etc/nginx . 2>/dev/null
            fi
            if [ -d /etc/letsencrypt ]; then
                tar -czf "$backup_path/ssl.tar.gz" -C /etc/letsencrypt . 2>/dev/null
            fi
            print_msg "$GREEN" "✅ Configurations backed up"
            ;;
        *)
            print_msg "$RED" "❌ Invalid option"
            rm -rf "$backup_path"
            echo
            read -p "Press Enter to continue..."
            return
            ;;
    esac
    
    # Create info file
    cat > "$backup_path/backup_info.txt" <<EOF
=================================
Backup Information
=================================
Date: $(date)
Server: $(hostname)
IP: $(hostname -I | awk '{print $1}')
Backup Type: Option $backup_choice
=================================
EOF
    
    # Compress everything
    print_msg "$YELLOW" "🗜️  Compressing backup..."
    cd "$BACKUP_DIR"
    tar -czf "${backup_name}.tar.gz" "$backup_name" 2>/dev/null
    rm -rf "$backup_name"
    
    if [ -f "${backup_name}.tar.gz" ]; then
        local backup_size=$(du -h "${backup_name}.tar.gz" 2>/dev/null | cut -f1)
        print_msg "$GREEN" "✅ Backup created successfully!"
        print_msg "$CYAN" "📍 Location: $BACKUP_DIR/${backup_name}.tar.gz"
        print_msg "$CYAN" "📊 Size: $backup_size"
        
        # Clean old backups (keep last 7)
        print_msg "$YELLOW" "🧹 Cleaning old backups..."
        ls -t "$BACKUP_DIR"/backup_*.tar.gz 2>/dev/null | tail -n +8 | xargs rm -f 2>/dev/null
        
        log_action "Backup created: ${backup_name}.tar.gz"
    else
        print_msg "$RED" "❌ Backup creation failed"
    fi
    
    echo
    read -p "Press Enter to continue..."
}

# Restore backup
restore_backup() {
    print_header
    print_msg "$CYAN" "♻️  Restore Backup"
    echo
    
    print_msg "$YELLOW" "📋 Available backups:"
    local backups=($(ls -t "$BACKUP_DIR"/backup_*.tar.gz 2>/dev/null))
    
    if [ ${#backups[@]} -eq 0 ]; then
        print_msg "$RED" "❌ No backups found!"
        echo
        read -p "Press Enter to continue..."
        return
    fi
    
    local i=1
    for backup in "${backups[@]}"; do
        local backup_name=$(basename "$backup")
        local backup_date=$(echo "$backup_name" | sed 's/backup_\(.*\)\.tar\.gz/\1/')
        local backup_size=$(du -h "$backup" 2>/dev/null | cut -f1)
        echo -e "${YELLOW}$i)${NC} $backup_date (${backup_size})"
        ((i++))
    done
    echo -e "${YELLOW}0)${NC} Cancel"
    echo
    
    read -p "Select backup to restore: " backup_choice
    
    if [ "$backup_choice" = "0" ]; then
        return
    fi
    
    if [ "$backup_choice" -gt 0 ] && [ "$backup_choice" -le ${#backups[@]} ]; then
        local selected_backup="${backups[$((backup_choice-1))]}"
        
        print_msg "$RED" "⚠️  WARNING: This will replace current files!"
        read -p "Are you sure? (yes/no): " confirm
        
        if [ "$confirm" = "yes" ]; then
            print_msg "$BLUE" "🔄 Restoring backup..."
            
            local temp_dir="/tmp/restore_temp_$"
            mkdir -p "$temp_dir"
            
            if tar -xzf "$selected_backup" -C "$temp_dir" 2>/dev/null; then
                local backup_folder=$(ls "$temp_dir" 2>/dev/null | head -1)
                
                if [ -n "$backup_folder" ]; then
                    if [ -f "$temp_dir/$backup_folder/website.tar.gz" ]; then
                        print_msg "$YELLOW" "📁 Restoring website files..."
                        tar -xzf "$temp_dir/$backup_folder/website.tar.gz" -C /var/www 2>/dev/null && print_msg "$GREEN" "✅ Website restored"
                    fi
                    
                    if [ -f "$temp_dir/$backup_folder/nginx.tar.gz" ]; then
                        print_msg "$YELLOW" "⚙️  Restoring Nginx config..."
                        tar -xzf "$temp_dir/$backup_folder/nginx.tar.gz" -C /etc/nginx 2>/dev/null && print_msg "$GREEN" "✅ Nginx restored"
                    fi
                    
                    if nginx -t 2>/dev/null; then
                        systemctl restart nginx 2>/dev/null
                        print_msg "$GREEN" "✅ Restore completed successfully!"
                        log_action "Backup restored: $(basename $selected_backup)"
                    else
                        print_msg "$RED" "❌ Nginx configuration error!"
                    fi
                else
                    print_msg "$RED" "❌ Invalid backup structure"
                fi
            else
                print_msg "$RED" "❌ Failed to extract backup"
            fi
            
            rm -rf "$temp_dir"
        else
            print_msg "$YELLOW" "⚠️  Restore cancelled"
        fi
    else
        print_msg "$RED" "❌ Invalid selection!"
    fi
    
    echo
    read -p "Press Enter to continue..."
}

# Monitor resources
monitor_resources() {
    print_header
    print_msg "$CYAN" "📊 Real-time Resource Monitoring (Press Ctrl+C to exit)"
    echo
    
    while true; do
        local cpu=$(top -bn1 2>/dev/null | grep "Cpu(s)" | sed "s/.*, *\([0-9.]*\)%* id.*/\1/" | awk '{print 100 - $1}' || echo "0")
        local mem_used=$(free -m 2>/dev/null | awk 'NR==2{print $3}' || echo "0")
        local mem_total=$(free -m 2>/dev/null | awk 'NR==2{print $2}' || echo "0")
        local mem_percent=$(free 2>/dev/null | awk 'NR==2{printf "%.1f", ($3/$2) * 100}' || echo "0")
        local disk_used=$(df -h / 2>/dev/null | awk 'NR==2{print $3}' || echo "0")
        local disk_total=$(df -h / 2>/dev/null | awk 'NR==2{print $2}' || echo "0")
        local disk_percent=$(df / 2>/dev/null | awk 'NR==2{print $5}' | sed 's/%//' || echo "0")
        local connections=$(ss -ant 2>/dev/null | grep -c ESTAB || echo "0")
        
        tput cup 5 0
        echo -e "${GREEN}🖥️  CPU Usage:${NC}     ${cpu}%        "
        echo -e "${GREEN}💾 RAM Usage:${NC}     ${mem_used}MB / ${mem_total}MB (${mem_percent}%)        "
        echo -e "${GREEN}💿 Disk Usage:${NC}    ${disk_used} / ${disk_total} (${disk_percent}%)        "
        echo -e "${GREEN}🌐 Connections:${NC}  ${connections}        "
        echo "                                                          "
        
        sleep 2
    done
}

#!/bin/bash

# =======================================================
# Rezgee Project Auto-Update Script from GitHub
# Updates the project automatically from GitHub repository
# =======================================================

set -e  # Stop on any error

# Colors for messages
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Basic settings
PROJECT_PATH="/var/www/rezgee"
BACKUP_DIR="/var/backups/rezgee"
GIT_BRANCH="main"  # or master depending on your branch name
LOG_FILE="/var/log/rezgee-update.log"
TIMESTAMP=$(date +"%Y%m%d_%H%M%S")

# Function to print messages
print_message() {
    local color=$1
    local message=$2
    echo -e "${color}${message}${NC}"
    
    # Create log directory if it doesn't exist
    local log_dir=$(dirname "$LOG_FILE")
    if [ ! -d "$log_dir" ]; then
        mkdir -p "$log_dir" 2>/dev/null || true
    fi
    
    # Log to file (silently fail if can't write)
    if [ -w "$log_dir" ] || [ -w "$LOG_FILE" ]; then
        echo "[$(date '+%Y-%m-%d %H:%M:%S')] ${message}" >> "$LOG_FILE" 2>/dev/null || true
    fi
}

# Function to check for errors
check_error() {
    local exit_code=$?
    if [ $exit_code -ne 0 ]; then
        print_message "$RED" "❌ Error: $1 (Exit code: $exit_code)"
        print_message "$YELLOW" "💡 Check logs for details: $LOG_FILE"
        exit 1
    fi
}

# Function to create backup
create_backup() {
    print_message "$BLUE" "📦 Creating backup..."
    
    # Check if project path exists
    if [ ! -d "$PROJECT_PATH" ]; then
        print_message "$RED" "❌ Error: Project path not found: $PROJECT_PATH"
        return 1
    fi
    
    # Create backup directory if it doesn't exist
    if [ ! -d "$BACKUP_DIR" ]; then
        print_message "$YELLOW" "Creating backup directory: $BACKUP_DIR"
        mkdir -p "$BACKUP_DIR" 2>/dev/null
        if [ $? -ne 0 ]; then
            print_message "$YELLOW" "⚠️  Cannot create $BACKUP_DIR, trying /tmp..."
            BACKUP_DIR="/tmp/rezgee-backups"
            mkdir -p "$BACKUP_DIR"
        fi
    fi
    
    # Check write permissions
    if [ ! -w "$BACKUP_DIR" ]; then
        print_message "$YELLOW" "⚠️  No write permission for $BACKUP_DIR, using /tmp..."
        BACKUP_DIR="/tmp/rezgee-backups"
        mkdir -p "$BACKUP_DIR"
    fi
    
    # Create compressed backup
    BACKUP_FILE="$BACKUP_DIR/rezgee_backup_$TIMESTAMP.tar.gz"
    print_message "$YELLOW" "Creating backup file: $BACKUP_FILE"
    
    cd "$PROJECT_PATH" || return 1
    tar -czf "$BACKUP_FILE" \
        --exclude='node_modules' \
        --exclude='.git' \
        --exclude='dist' \
        --exclude='build' \
        --exclude='.next' \
        --exclude='coverage' \
        . 2>&1 | head -10
    
    if [ -f "$BACKUP_FILE" ]; then
        BACKUP_SIZE=$(du -h "$BACKUP_FILE" | cut -f1)
        print_message "$GREEN" "✅ Backup created: $BACKUP_FILE ($BACKUP_SIZE)"
        
        # Delete old backups (keep last 5)
        cd "$BACKUP_DIR"
        OLD_BACKUPS=$(ls -t rezgee_backup_*.tar.gz 2>/dev/null | tail -n +6)
        if [ -n "$OLD_BACKUPS" ]; then
            echo "$OLD_BACKUPS" | xargs rm -f
            print_message "$YELLOW" "🗑️  Old backups removed"
        fi
        return 0
    else
        print_message "$YELLOW" "⚠️  Warning: Failed to create backup"
        print_message "$YELLOW" "Continuing without backup..."
        return 0  # Don't fail the entire update
    fi
}

# Function to check for updates
check_updates() {
    print_message "$BLUE" "🔍 Checking for new updates..."
    
    cd "$PROJECT_PATH" || return 1
    
    # Test GitHub connection first
    print_message "$YELLOW" "Testing GitHub connection..."
    if ! git ls-remote origin &>/dev/null; then
        print_message "$RED" "❌ Cannot connect to GitHub"
        print_message "$YELLOW" "💡 Possible issues:"
        print_message "$YELLOW" "   1. No internet connection"
        print_message "$YELLOW" "   2. SSH key not configured"
        print_message "$YELLOW" "   3. Wrong repository URL"
        print_message "$YELLOW" "   4. GitHub access denied"
        print_message "$YELLOW" ""
        print_message "$YELLOW" "Current remote URL:"
        git remote -v | grep origin
        return 1
    fi
    
    # Fetch latest changes without merging
    print_message "$YELLOW" "Fetching updates from branch: $GIT_BRANCH..."
    if ! git fetch origin "$GIT_BRANCH" 2>&1 | tee -a "$LOG_FILE"; then
        print_message "$RED" "❌ Failed to fetch updates"
        return 1
    fi
    
    # Check if updates exist
    LOCAL=$(git rev-parse HEAD 2>/dev/null)
    REMOTE=$(git rev-parse origin/"$GIT_BRANCH" 2>/dev/null)
    
    if [ -z "$LOCAL" ] || [ -z "$REMOTE" ]; then
        print_message "$RED" "❌ Cannot determine version information"
        return 1
    fi
    
    if [ "$LOCAL" = "$REMOTE" ]; then
        print_message "$GREEN" "✅ Project is up to date - no new updates"
        print_message "$YELLOW" "Current commit: ${LOCAL:0:8}"
        return 1
    else
        print_message "$YELLOW" "📥 New updates found"
        print_message "$YELLOW" "Local:  ${LOCAL:0:8}"
        print_message "$YELLOW" "Remote: ${REMOTE:0:8}"
        
        # Show what changed
        print_message "$YELLOW" ""
        print_message "$YELLOW" "Changes since last update:"
        git log --oneline HEAD..origin/"$GIT_BRANCH" | head -5
        
        return 0
    fi
}

# Function to pull updates
pull_updates() {
    print_message "$BLUE" "📥 Pulling updates from GitHub..."
    
    cd "$PROJECT_PATH"
    
    # Check for uncommitted local changes
    if [ -n "$(git status --porcelain)" ]; then
        print_message "$YELLOW" "⚠️  Local changes detected - stashing them temporarily"
        git stash
        check_error "Failed to stash changes"
    fi
    
    # Pull updates
    git pull origin "$GIT_BRANCH"
    check_error "Failed to pull updates"
    
    print_message "$GREEN" "✅ Updates pulled successfully"
}

# Function to install dependencies
install_dependencies() {
    print_message "$BLUE" "📦 Updating dependencies..."
    
    cd "$PROJECT_PATH"
    
    # Check for package.json
    if [ -f "package.json" ]; then
        # Install Node.js dependencies
        if command -v npm &> /dev/null; then
            print_message "$YELLOW" "📦 Installing npm dependencies (including dev dependencies for build)..."
            
            # Use npm ci if package-lock.json exists (faster and cleaner)
            if [ -f "package-lock.json" ]; then
                print_message "$YELLOW" "Using npm ci for clean install..."
                npm ci
            else
                print_message "$YELLOW" "Using npm install..."
                npm install
            fi
            
            check_error "Failed to install npm dependencies"
            print_message "$GREEN" "✅ npm dependencies installed"
        fi
        
        # Install pnpm dependencies if exists
        if command -v pnpm &> /dev/null && [ -f "pnpm-lock.yaml" ]; then
            print_message "$YELLOW" "📦 Installing pnpm dependencies..."
            pnpm install
            check_error "Failed to install pnpm dependencies"
            print_message "$GREEN" "✅ pnpm dependencies installed"
        fi
    fi
    
    # Check for requirements.txt (Python)
    if [ -f "requirements.txt" ]; then
        if command -v pip3 &> /dev/null; then
            print_message "$YELLOW" "📦 Installing Python dependencies..."
            pip3 install -r requirements.txt
            check_error "Failed to install Python dependencies"
            print_message "$GREEN" "✅ Python dependencies installed"
        fi
    fi
}

# Function to build project
build_project() {
    print_message "$BLUE" "🔨 Building project..."
    
    cd "$PROJECT_PATH"
    
    # Build project if it has a build command
    if [ -f "package.json" ] && grep -q '"build"' package.json; then
        if command -v npm &> /dev/null; then
            print_message "$YELLOW" "🔨 Running npm run build..."
            
            # Set NODE_ENV to production for optimized build
            export NODE_ENV=production
            
            # Run build with error output
            if npm run build 2>&1 | tee -a "$LOG_FILE"; then
                print_message "$GREEN" "✅ Project built successfully"
            else
                print_message "$RED" "❌ Build failed"
                print_message "$YELLOW" "💡 Common fixes:"
                print_message "$YELLOW" "   1. Make sure all dependencies are installed: npm install"
                print_message "$YELLOW" "   2. Check if vite/build tool is in devDependencies"
                print_message "$YELLOW" "   3. Try: npm install && npm run build"
                print_message "$YELLOW" "   4. Check .env file exists"
                return 1
            fi
        elif command -v pnpm &> /dev/null; then
            print_message "$YELLOW" "🔨 Running pnpm build..."
            export NODE_ENV=production
            if pnpm build 2>&1 | tee -a "$LOG_FILE"; then
                print_message "$GREEN" "✅ Project built successfully"
            else
                print_message "$RED" "❌ Build failed"
                return 1
            fi
        fi
    else
        print_message "$YELLOW" "ℹ️  No build command found in package.json, skipping..."
    fi
}

# Function to restore from backup
restore_backup() {
    if [ -z "$1" ]; then
        print_message "$RED" "❌ Backup file must be specified"
        echo "Usage: $0 --restore /path/to/backup.tar.gz"
        exit 1
    fi
    
    BACKUP_FILE="$1"
    
    if [ ! -f "$BACKUP_FILE" ]; then
        print_message "$RED" "❌ Backup file not found: $BACKUP_FILE"
        exit 1
    fi
    
    print_message "$YELLOW" "⚠️  Are you sure you want to restore from backup? (yes/no)"
    read -r confirmation
    
    if [ "$confirmation" != "yes" ]; then
        print_message "$YELLOW" "❌ Cancelled"
        exit 0
    fi
    
    print_message "$BLUE" "📦 Restoring from backup..."
    
    # Create backup of current state before restoring
    create_backup
    
    # Restore files
    rm -rf "$PROJECT_PATH"/*
    tar -xzf "$BACKUP_FILE" -C "$PROJECT_PATH"
    check_error "Failed to restore from backup"
    
    print_message "$GREEN" "✅ Backup restored successfully"
    
    # Restart services
    restart_services
}

# Function to list available backups
list_backups() {
    print_message "$BLUE" "📦 Available backups:"
    echo ""
    
    if [ ! -d "$BACKUP_DIR" ]; then
        print_message "$YELLOW" "No backup directory found"
        return
    fi
    
    cd "$BACKUP_DIR"
    
    if [ -z "$(ls -A)" ]; then
        print_message "$YELLOW" "No backups found"
        return
    fi
    
    # List backups with details
    ls -lth rezgee_backup_*.tar.gz 2>/dev/null | awk '{

    }' | head -20
    
    echo ""
    print_message "$GREEN" "Total backups: $(ls -1 rezgee_backup_*.tar.gz 2>/dev/null | wc -l)"
}

    echo ""
    
    # Check project path
    print_message "$YELLOW" "1️⃣  Checking project path..."
    if [ -d "$PROJECT_PATH" ]; then
        print_message "$GREEN" "   ✅ Project path exists: $PROJECT_PATH"

    else
        print_message "$RED" "   ❌ Project path not found: $PROJECT_PATH"
    fi
    echo ""
    
    # Check Git
    print_message "$YELLOW" "2️⃣  Checking Git repository..."
    if [ -d "$PROJECT_PATH/.git" ]; then
        print_message "$GREEN" "   ✅ Git repository exists"
        cd "$PROJECT_PATH"
        echo "   Current branch: $(git branch --show-current 2>/dev/null || echo 'unknown')"
        echo "   Last commit: $(git log -1 --oneline 2>/dev/null || echo 'unknown')"
        echo "   Remote URL:"
        git remote -v 2>/dev/null || echo "   No remote configured"
    else
        print_message "$RED" "   ❌ Not a Git repository"
    fi
    echo ""
    
    # Check GitHub connection
    print_message "$YELLOW" "3️⃣  Checking GitHub connection..."
    cd "$PROJECT_PATH"
    if git ls-remote origin &>/dev/null; then
        print_message "$GREEN" "   ✅ Can connect to GitHub"
    else
        print_message "$RED" "   ❌ Cannot connect to GitHub"
        print_message "$YELLOW" "   Trying to fetch..."
        git fetch origin 2>&1 | head -5
    fi

    print_message "$YELLOW" "4️⃣  Checking backup directory..."
    if [ -d "$BACKUP_DIR" ]; then

        echo "   Backups count: $(ls -1 "$BACKUP_DIR"/rezgee_backup_*.tar.gz 2>/dev/null | wc -l)"
        echo "   Free space: $(df -h "$BACKUP_DIR" | awk 'NR==2 {print $4}')"

        print_message "$YELLOW" "   Will be created automatically"
    fi
    echo ""
    
    # Check permissions
    print_message "$YELLOW" "5️⃣  Checking permissions..."
    echo "   Current user: $(whoami)"
    echo "   Project owner: $(ls -ld "$PROJECT_PATH" | awk '{print $3":"$4}')"
    if [ -w "$PROJECT_PATH" ]; then
        print_message "$GREEN" "   ✅ Can write to project directory"
    else
        print_message "$RED" "   ❌ Cannot write to project directory"
    fi
    echo ""
    
    # Check services
    print_message "$YELLOW" "6️⃣  Checking services..."
    if command -v pm2 &> /dev/null; then
        print_message "$GREEN" "   ✅ PM2 installed"
    else
        print_message "$YELLOW" "   ⚠️  PM2 not installed"
    fi

        print_message "$GREEN" "   ✅ Nginx installed"
    else
        print_message "$YELLOW" "   ⚠️  Nginx not installed"

        print_message "$GREEN" "   ✅ npm installed ($(npm -v))"
    else
        print_message "$YELLOW" "   ⚠️  npm not installed"
    fi

    print_message "$YELLOW" "7️⃣  Checking log file..."
    if [ -f "$LOG_FILE" ]; then

# Main function
main() {
    # Check for help first
    if [ "$1" = "--help" ] || [ "$1" = "-h" ]; then
        show_help
        exit 0
    fi
    
    print_message "$BLUE" "═══════════════════════════════════════════"
    print_message "$BLUE" "    🚀 Rezgee Project Update Script"
    print_message "$BLUE" "═══════════════════════════════════════════"
    echo ""
    
    # Check access permissions
    if [ ! -d "$PROJECT_PATH" ]; then
        print_message "$RED" "❌ Project path not found: $PROJECT_PATH"
        print_message "$YELLOW" "💡 Please check if the project path is correct"
        exit 1
    fi
    
    if [ ! -d "$PROJECT_PATH/.git" ]; then
        print_message "$RED" "❌ Project is not a Git repository"
        print_message "$YELLOW" "💡 Initialize git repository first:"
        print_message "$YELLOW" "   cd $PROJECT_PATH"
        print_message "$YELLOW" "   git init"
        print_message "$YELLOW" "   git remote add origin <your-repo-url>"
        exit 1
    fi
    
    # Test git access
    cd "$PROJECT_PATH" || exit 1
    if ! git status &>/dev/null; then
        print_message "$RED" "❌ Cannot access git repository"
        print_message "$YELLOW" "💡 Please check git permissions"
        exit 1
    fi
    
    # Handle options
    case "${1:-}" in
        --status)
            show_status
            exit 0
            ;;
        --diagnose|--debug)
            diagnose_system
            exit 0
            ;;
        --backup-only)
            create_backup
            exit 0
            ;;
        --list-backups)
            list_backups
            exit 0
            ;;
        --restore)
            restore_backup "$2"
            exit 0
            ;;
        --no-backup)
            SKIP_BACKUP=true
            ;;
        --force)
            FORCE_UPDATE=true
            ;;
        --update|"")
            # Default action - update
            ;;
        *)
            print_message "$RED" "❌ Unknown option: $1"
            echo ""
            show_help
            exit 1
            ;;
    esac
    
    # Create backup (unless skipped)
    if [ "${SKIP_BACKUP:-false}" != "true" ]; then
        create_backup
        if [ $? -ne 0 ]; then
            print_message "$YELLOW" "⚠️  Backup creation had issues but continuing..."
        fi
    fi
    
    # Check for updates
    UPDATES_AVAILABLE=false
    if check_updates || [ "${FORCE_UPDATE:-false}" = "true" ]; then
        UPDATES_AVAILABLE=true
    fi
    
    # Pull updates only if available
    if [ "$UPDATES_AVAILABLE" = true ]; then
        print_message "$YELLOW" "📥 New updates detected - pulling changes..."
        pull_updates
        if [ $? -ne 0 ]; then
            print_message "$RED" "❌ Failed to pull updates"
            exit 1
        fi
    else
        print_message "$GREEN" "═══════════════════════════════════════════"
        print_message "$GREEN" "    ✅ Project is already up to date!"
        print_message "$GREEN" "═══════════════════════════════════════════"
        print_message "$BLUE" ""
        print_message "$BLUE" "ℹ️  Skipping git pull, continuing with other operations..."
        print_message "$BLUE" ""
    fi
    
    # Continue with remaining operations (regardless of updates)
    # Install dependencies
    install_dependencies
    if [ $? -ne 0 ]; then
        print_message "$RED" "❌ Failed to install dependencies"
        exit 1
    fi
    
    # Build project
    build_project
    if [ $? -ne 0 ]; then
        print_message "$RED" "❌ Failed to build project"
        print_message "$YELLOW" "💡 Project updated but build failed"
        print_message "$YELLOW" "💡 You may need to build manually:"
        print_message "$YELLOW" "   cd $PROJECT_PATH"
        print_message "$YELLOW" "   npm install"
        print_message "$YELLOW" "   npm run build"
        exit 1
    fi
    
    # Cleanup
    cleanup
    
    # Restart services
    restart_services
    
    # Show version info
    show_version_info
    
    # Final success message
    print_message "$GREEN" "═══════════════════════════════════════════"
    if [ "$UPDATES_AVAILABLE" = true ]; then
        print_message "$GREEN" "    ✅ Project updated successfully!"
    else
        print_message "$GREEN" "    ✅ Project rebuilt successfully!"
    fi
    print_message "$GREEN" "═══════════════════════════════════════════"
}

# Run main function
main "$@"
